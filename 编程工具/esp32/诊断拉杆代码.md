```C++
// --- 电机引脚 ---
#define RPWM_PIN 18 
#define LPWM_PIN 19

// --- 电流检测引脚 ---
#define R_IS_PIN 34 // R_IS (开门)
#define L_IS_PIN 35 // L_IS (关门)

void setup() {
  Serial.begin(115200);
  Serial.println("电流诊断医生 v2.0");
  Serial.println("请输入 'open' 或 'close' 来启动电机并观察电流...");

  pinMode(RPWM_PIN, OUTPUT);
  pinMode(LPWM_PIN, OUTPUT);
  digitalWrite(RPWM_PIN, LOW);
  digitalWrite(LPWM_PIN, LOW); // 确保电机初始状态是停止
}

void loop() {
  if (Serial.available() > 0) {
    String command = Serial.readStringUntil('\n');
    command.trim();

    if (command == "open") {
      // 执行开门，并持续监控开门电流
      Serial.println("\n--- 开始开门 | 监控 R_IS 电流 ---");
      digitalWrite(RPWM_PIN, HIGH);
      digitalWrite(LPWM_PIN, LOW);
      
      // 监控5秒钟，足够让它从启动到堵转
      monitorCurrent(R_IS_PIN, 5000); 
      
      digitalWrite(RPWM_PIN, LOW); // 结束后停止
      Serial.println("--- 开门监控结束 ---\n");

    } else if (command == "close") {
      // 执行关门，并持续监控关门电流
      Serial.println("\n--- 开始关门 | 监控 L_IS 电流 ---");
      digitalWrite(RPWM_PIN, LOW);
      digitalWrite(LPWM_PIN, HIGH);
      
      // 监控5秒钟
      monitorCurrent(L_IS_PIN, 5000);
      
      digitalWrite(LPWM_PIN, LOW); // 结束后停止
      Serial.println("--- 关门监控结束 ---\n");
    }
  }
}

// 这是一个辅助函数，专门用来在指定时间内高频打印指定引脚的电流值
void monitorCurrent(int pin, int duration_ms) {
  unsigned long startTime = millis();
  while (millis() - startTime < duration_ms) {
    int current = analogRead(pin);
    Serial.println(current);
    delay(50); // 每50毫秒打印一次，看得更清楚
  }
}
```
